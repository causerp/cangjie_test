enum TimeUnit{
Month |
Year(Int64)|
Day(Int64, Int64)
}

main(){
var a1 = Year(1)
var b = match (a1) {
  case TimeUnit.Month => 3
  case TimeUnit.Year(1) => 4
  case TimeUnit.Year(_) => 5
  case TimeUnit.Day(_, _) => 6
  }

  var a2 = Day(0, 2)
var b2 = match (a2) {
  case TimeUnit.Month => 3
  case TimeUnit.Year(1) => 4
  case TimeUnit.Year(_) => 5
  case TimeUnit.Day(_, _) => 6
  }
  return b + b2 - 10
}
